# plot the vectors v, w and v+W in figure

# Create vectors 'v' and 'w'
v = np.array([5, 6])
w = np.array([7, 8])

# Calculate the sum v + w
v_plus_w = v + w

# Plot the vectors
origin = np.zeros_like(v)

plt.quiver(*origin, *v, color='r', scale=1, scale_units='xy', angles='xy', label='v')
plt.quiver(*origin, *w, color='b', scale=1, scale_units='xy', angles='xy', label='w')
plt.quiver(*origin, *v_plus_w, color='g', scale=1, scale_units='xy', angles='xy', label='v + w')

# Set axis limits
plt.xlim(0, max(v[0], w[0], v_plus_w[0]) + 1)
plt.ylim(0, max(v[1], w[1], v_plus_w[1]) + 1)

# Add labels and legend
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.legend()

# Show the plot
plt.grid()
plt.show()
