# Read the data 
import seaborn as sns
import pandas as pd
from sklearn.preprocessing import StandardScaler, MinMaxScaler, RobustScaler
import matplotlib.pyplot as plt

# Load the Iris dataset
iris = sns.load_dataset('iris')

# Display the first few rows of the dataset
print("Original Iris Dataset:")
print(iris.head())

# Extract the 'sepal_length' column for scaling
sepal_length_data = iris[['sepal_length']]

# Scale the data using StandardScaler
standard_scaler = StandardScaler()
sepal_length_standard_scaled = standard_scaler.fit_transform(sepal_length_data)

# Scale the data using MinMaxScaler
minmax_scaler = MinMaxScaler()
sepal_length_minmax_scaled = minmax_scaler.fit_transform(sepal_length_data)

# Scale the data using RobustScaler
robust_scaler = RobustScaler()
sepal_length_robust_scaled = robust_scaler.fit_transform(sepal_length_data)

# Convert the scaled arrays back to a DataFrame for better visualization
sepal_length_standard_scaled_df = pd.DataFrame(sepal_length_standard_scaled, columns=['sepal_length_standard_scaled'])
sepal_length_minmax_scaled_df = pd.DataFrame(sepal_length_minmax_scaled, columns=['sepal_length_minmax_scaled'])
sepal_length_robust_scaled_df = pd.DataFrame(sepal_length_robust_scaled, columns=['sepal_length_robust_scaled'])

# Concatenate the scaled DataFrames along columns for easy comparison
scaled_data_comparison = pd.concat([sepal_length_data,
                                    sepal_length_standard_scaled_df,
                                    sepal_length_minmax_scaled_df,
                                    sepal_length_robust_scaled_df],
                                   axis=1)

# Display the first few rows of the scaled data
print("\nScaled Sepal Length Data:")
print(scaled_data_comparison.head())

# Visualize the original and scaled sepal_length data
plt.figure(figsize=(10, 6))
sns.kdeplot(iris['sepal_length'], label='Original Sepal Length', linewidth=2)
sns.kdeplot(sepal_length_standard_scaled.flatten(), label='Standard Scaled', linewidth=2)
sns.kdeplot(sepal_length_minmax_scaled.flatten(), label='MinMax Scaled', linewidth=2)
sns.kdeplot(sepal_length_robust_scaled.flatten(), label='Robust Scaled', linewidth=2)
plt.title('Sepal Length Scaling Comparison')
plt.xlabel('Sepal Length')
plt.ylabel('Density')
plt.legend()
plt.show()
